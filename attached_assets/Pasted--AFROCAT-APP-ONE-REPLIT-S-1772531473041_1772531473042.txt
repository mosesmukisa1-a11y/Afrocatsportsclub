import fs from "fs";
import path from "path";

const dataDir = path.resolve("data");
if (!fs.existsSync(dataDir)) fs.mkdirSync(dataDir, { recursive: true });

function readJson(name, fallback) {
  const p = path.join(dataDir, name);
  if (!fs.existsSync(p)) return fallback;
  try { return JSON.parse(fs.readFileSync(p, "utf-8") || ""); } catch { return fallback; }
}
function writeJson(name, value) {
  const p = path.join(dataDir, name);
  fs.writeFileSync(p, JSON.stringify(value, null, 2), "utf-8");
}
function uid(prefix="ID") {
  return `${prefix}_${Date.now()}_${Math.random().toString(16).slice(2)}`;
}

let state = {
  players: readJson("players.json", []),          // {id, firstName,lastName,dob,jerseyNumber,teamId,photoUrl}
  teams: readJson("teams.json", []),              // {id,name}
  matches: readJson("matches.json", []),          // see match fields in routes
  matchEvents: readJson("matchEvents.json", []),  // {id,matchId,teamId,playerId,action,outcome,createdAt,createdBy}
  matchStats: readJson("matchStats.json", []),    // {matchId,sets,enteredBy,enteredAt}
  notifications: readJson("notifications.json", []), // {id,targetUserId|null,title,message,type,dateKey,matchId,reminderType,createdAt}
  attendanceSessions: readJson("attendanceSessions.json", []), // {id,teamId,sessionDate,sessionType,status,lockedAt,lockedBy,createdBy,createdAt}
  attendanceLines: readJson("attendanceLines.json", []), // {id,attendanceSessionId,playerId,present,note}
  contractAcceptances: readJson("contractAcceptances.json", []), // see contract route
};

function persist() {
  writeJson("players.json", state.players);
  writeJson("teams.json", state.teams);
  writeJson("matches.json", state.matches);
  writeJson("matchEvents.json", state.matchEvents);
  writeJson("matchStats.json", state.matchStats);
  writeJson("notifications.json", state.notifications);
  writeJson("attendanceSessions.json", state.attendanceSessions);
  writeJson("attendanceLines.json", state.attendanceLines);
  writeJson("contractAcceptances.json", state.contractAcceptances);
}

export const db = {
  // TEAM
  team: {
    findUnique: async ({ where: { id } }) => state.teams.find(t => t.id === id) || null,
  },
  // PLAYER
  player: {
    findMany: async ({ where = {}, select, orderBy } = {}) => {
      let rows = [...state.players];
      if (where.teamId) rows = rows.filter(p => p.teamId === where.teamId);
      if (orderBy?.length) {
        for (const ob of orderBy.reverse()) {
          const [k, dir] = Object.entries(ob)[0];
          rows.sort((a,b) => (a[k] ?? "").toString().localeCompare((b[k] ?? "").toString()) * (dir === "desc" ? -1 : 1));
        }
      }
      if (select) {
        rows = rows.map(r => {
          const o = {};
          Object.keys(select).forEach(k => { if (select[k]) o[k] = r[k]; });
          return o;
        });
      }
      return rows;
    },
    findUnique: async ({ where: { id } }) => state.players.find(p => p.id === id) || null,
  },
  // MATCH
  match: {
    findUnique: async ({ where: { id } }) => state.matches.find(m => m.id === id) || null,
    findMany: async ({ where = {}, include } = {}) => {
      let rows = [...state.matches];
      if (where.status) rows = rows.filter(m => m.status === where.status);
      if (where.startTime?.gte) rows = rows.filter(m => new Date(m.startTime) >= new Date(where.startTime.gte));
      if (where.startTime?.lte) rows = rows.filter(m => new Date(m.startTime) <= new Date(where.startTime.lte));
      if (include?.homeTeam || include?.awayTeam) {
        rows = rows.map(m => ({
          ...m,
          homeTeam: state.teams.find(t => t.id === m.homeTeamId) || null,
          awayTeam: state.teams.find(t => t.id === m.awayTeamId) || null,
        }));
      }
      return rows;
    },
    create: async ({ data }) => {
      const row = { id: uid("MATCH"), ...data };
      state.matches.unshift(row); persist(); return row;
    },
    update: async ({ where: { id }, data }) => {
      const idx = state.matches.findIndex(m => m.id === id);
      if (idx < 0) return null;
      state.matches[idx] = { ...state.matches[idx], ...data };
      persist(); return state.matches[idx];
    },
  },
  // MATCH STATS
  matchStats: {
    upsert: async ({ where: { matchId }, update, create }) => {
      const idx = state.matchStats.findIndex(s => s.matchId === matchId);
      if (idx >= 0) {
        state.matchStats[idx] = { ...state.matchStats[idx], ...update };
        persist(); return state.matchStats[idx];
      }
      state.matchStats.unshift({ ...create });
      persist(); return state.matchStats[0];
    },
    findUnique: async ({ where: { matchId } }) => state.matchStats.find(s => s.matchId === matchId) || null,
  },
  // MATCH EVENTS
  matchEvent: {
    create: async ({ data }) => {
      const row = { id: uid("EVT"), ...data };
      state.matchEvents.unshift(row); persist(); return row;
    },
    delete: async ({ where: { id } }) => {
      state.matchEvents = state.matchEvents.filter(e => e.id !== id);
      persist(); return true;
    },
  },
  // NOTIFICATIONS
  notification: {
    create: async ({ data }) => {
      const row = { id: uid("NTF"), createdAt: new Date().toISOString(), ...data };
      state.notifications.unshift(row); persist(); return row;
    },
    findMany: async ({ where = {}, orderBy, take } = {}) => {
      let rows = [...state.notifications];
      if (where.targetUserId === null) rows = rows.filter(n => n.targetUserId === null);
      if (where.type) rows = rows.filter(n => n.type === where.type);
      if (where.matchId) rows = rows.filter(n => n.matchId === where.matchId);
      if (where.reminderType) rows = rows.filter(n => n.reminderType === where.reminderType);
      if (where.dateKey) rows = rows.filter(n => n.dateKey === where.dateKey);
      if (orderBy?.createdAt === "desc") rows.sort((a,b)=> new Date(b.createdAt)-new Date(a.createdAt));
      if (typeof take === "number") rows = rows.slice(0, take);
      return rows;
    },
    findFirst: async ({ where = {} }) => {
      const rows = await db.notification.findMany({ where, take: 1 });
      return rows[0] || null;
    },
  },
  // ATTENDANCE
  attendanceSession: {
    findUnique: async ({ where: { id } }) => state.attendanceSessions.find(s => s.id === id) || null,
    update: async ({ where: { id }, data }) => {
      const idx = state.attendanceSessions.findIndex(s => s.id === id);
      if (idx < 0) return null;
      state.attendanceSessions[idx] = { ...state.attendanceSessions[idx], ...data };
      persist(); return state.attendanceSessions[idx];
    },
  },
  attendanceLine: {
    upsert: async ({ where: { attendanceSessionId_playerId }, update, create }) => {
      const { attendanceSessionId, playerId } = attendanceSessionId_playerId;
      const idx = state.attendanceLines.findIndex(l => l.attendanceSessionId === attendanceSessionId && l.playerId === playerId);
      if (idx >= 0) {
        state.attendanceLines[idx] = { ...state.attendanceLines[idx], ...update };
        persist(); return state.attendanceLines[idx];
      }
      const row = { id: uid("ATL"), ...create };
      state.attendanceLines.unshift(row);
      persist(); return row;
    },
  },
  // CONTRACT
  contractAcceptance: {
    findFirst: async ({ where: { userId, contractKey } }) =>
      state.contractAcceptances.find(a => a.userId === userId && a.contractKey === contractKey) || null,
    upsert: async ({ where: { userId_contractKey }, update, create }) => {
      const { userId, contractKey } = userId_contractKey;
      const idx = state.contractAcceptances.findIndex(a => a.userId === userId && a.contractKey === contractKey);
      if (idx >= 0) {
        state.contractAcceptances[idx] = { ...state.contractAcceptances[idx], ...update };
        persist(); return state.contractAcceptances[idx];
      }
      const row = { id: uid("CTA"), ...create };
      state.contractAcceptances.unshift(row);
      persist(); return row;
    },
  },
};

export function calcAge(dobISO) {
  if (!dobISO) return null;
  const dob = new Date(dobISO);
  if (Number.isNaN(dob.getTime())) return null;

  const today = new Date();
  let age = today.getFullYear() - dob.getFullYear();
  const m = today.getMonth() - dob.getMonth();
  if (m < 0 || (m === 0 && today.getDate() < dob.getDate())) age--;
  return age;
}
export function fmtDate(dateISO) {
  if (!dateISO) return "";
  const d = new Date(dateISO);
  if (Number.isNaN(d.getTime())) return "";
  const dd = String(d.getDate()).padStart(2,"0");
  const mm = String(d.getMonth()+1).padStart(2,"0");
  const yy = d.getFullYear();
  return `${dd}/${mm}/${yy}`;
}
export function fmtDateTime(dateISO) {
  if (!dateISO) return "";
  const d = new Date(dateISO);
  if (Number.isNaN(d.getTime())) return "";
  const dd = String(d.getDate()).padStart(2,"0");
  const mm = String(d.getMonth()+1).padStart(2,"0");
  const yy = d.getFullYear();
  const hh = String(d.getHours()).padStart(2,"0");
  const mi = String(d.getMinutes()).padStart(2,"0");
  return `${dd}/${mm}/${yy} ${hh}:${mi}`;
}

import express from "express";
import { requireRole } from "../utils/roles.js";

const router = express.Router();

function getMatchStatus(startTime) {
  const now = new Date();
  return new Date(startTime) > now ? "UPCOMING" : "SCHEDULED";
}
function isMatchLocked(match) {
  return !!match.statsEntered || !!match.scoreLocked || match.scoreSource === "STATS";
}
function computeScoreFromStats(sets) {
  let homeSets = 0, awaySets = 0;
  for (const s of sets) {
    if (s.homePoints > s.awayPoints) homeSets++;
    else if (s.awayPoints > s.homePoints) awaySets++;
  }
  return { homeScore: homeSets, awayScore: awaySets };
}

// CREATE MATCH (score NOT required for upcoming)
router.post("/matches", requireRole(["ADMIN","COACH","SUPER_ADMIN"]), async (req, res) => {
  const { startTime, homeTeamId, awayTeamId, venue = "" } = req.body;
  if (!startTime || !homeTeamId || !awayTeamId) {
    return res.status(400).json({ ok:false, error:"Missing required fields" });
  }

  const status = getMatchStatus(startTime);

  const match = await req.db.match.create({
    data: {
      startTime: new Date(startTime).toISOString(),
      homeTeamId,
      awayTeamId,
      venue,
      status,
      homeScore: null,
      awayScore: null,
      scoreSource: "NONE",
      scoreLocked: false,
      statsEntered: false,
      lastScoreUpdatedBy: null,
      lastScoreUpdatedAt: null,
    },
  });

  return res.json({ ok:true, match });
});

// ENTER SET STATS (locks score, marks PLAYED)
router.post("/matches/:id/stats", requireRole(["ADMIN","COACH","SUPER_ADMIN"]), async (req, res) => {
  const matchId = req.params.id;
  const { sets } = req.body;

  if (!Array.isArray(sets) || sets.length === 0) {
    return res.status(400).json({ ok:false, error:"Sets stats required" });
  }

  const match = await req.db.match.findUnique({ where: { id: matchId } });
  if (!match) return res.status(404).json({ ok:false, error:"Match not found" });

  if (isMatchLocked(match)) {
    return res.status(403).json({ ok:false, error:"Stats locked: already submitted." });
  }

  await req.db.matchStats.upsert({
    where: { matchId },
    update: { sets, enteredBy: req.user.id, enteredAt: new Date().toISOString() },
    create: { matchId, sets, enteredBy: req.user.id, enteredAt: new Date().toISOString() },
  });

  const { homeScore, awayScore } = computeScoreFromStats(sets);

  const updated = await req.db.match.update({
    where: { id: matchId },
    data: {
      homeScore,
      awayScore,
      scoreSource: "STATS",
      scoreLocked: true,
      statsEntered: true,
      status: "PLAYED",
      lastScoreUpdatedBy: req.user.id,
      lastScoreUpdatedAt: new Date().toISOString(),
    },
  });

  return res.json({ ok:true, match: updated });
});

// MANUAL SCORE (only if past AND no stats)
router.patch("/matches/:id/score", requireRole(["ADMIN","COACH","SUPER_ADMIN"]), async (req, res) => {
  const matchId = req.params.id;
  const { homeScore, awayScore } = req.body;

  const match = await req.db.match.findUnique({ where: { id: matchId } });
  if (!match) return res.status(404).json({ ok:false, error:"Match not found" });

  const now = new Date();
  if (new Date(match.startTime) > now) {
    return res.status(400).json({ ok:false, error:"Cannot set score for an upcoming match" });
  }

  if (match.statsEntered || match.scoreSource === "STATS" || match.scoreLocked) {
    return res.status(403).json({ ok:false, error:"Score locked: stats already entered" });
  }

  if (!Number.isInteger(homeScore) || !Number.isInteger(awayScore)) {
    return res.status(400).json({ ok:false, error:"Scores must be integers" });
  }

  const updated = await req.db.match.update({
    where: { id: matchId },
    data: {
      homeScore,
      awayScore,
      scoreSource: "MANUAL",
      scoreLocked: true,
      status: "PLAYED",
      lastScoreUpdatedBy: req.user.id,
      lastScoreUpdatedAt: new Date().toISOString(),
    },
  });

  return res.json({ ok:true, match: updated });
});

export default router;

export function createUpcomingMatchRemindersJob(cron, db) {
  async function createReminderIfMissing(match, reminderType, title, message, dateKey) {
    const existing = await db.notification.findFirst({
      where: { type: "MATCH_REMINDER", matchId: match.id, reminderType, dateKey },
    });
    if (existing) return;

    // targetUserId = null means "ALL relevant users" in this simple store
    await db.notification.create({
      data: {
        targetUserId: null,
        type: "MATCH_REMINDER",
        title,
        message,
        matchId: match.id,
        reminderType,
        dateKey,
      },
    });
  }

  function dayKey(d) {
    const dt = new Date(d);
    const y = dt.getFullYear();
    const m = String(dt.getMonth() + 1).padStart(2, "0");
    const dd = String(dt.getDate()).padStart(2, "0");
    return `${y}-${m}-${dd}`;
  }

  cron.schedule("*/15 * * * *", async () => {
    const now = new Date();
    const in48h = new Date(now.getTime() + 48 * 60 * 60 * 1000);

    const matches = await db.match.findMany({
      where: { status: "UPCOMING", startTime: { gte: now.toISOString(), lte: in48h.toISOString() } },
      include: { homeTeam: true, awayTeam: true },
    });

    for (const m of matches) {
      const start = new Date(m.startTime);
      const diffMs = start.getTime() - now.getTime();
      const diffH = diffMs / (60 * 60 * 1000);

      const dk = dayKey(now);

      const base = `${m.homeTeam?.name || "Home"} vs ${m.awayTeam?.name || "Away"} @ ${m.venue || "TBA"}`;
      if (diffH <= 48 && diffH > 24) {
        await createReminderIfMissing(m, "48H", "⏳ Match in 48 Hours", `${base}`, dk);
      } else if (diffH <= 24 && diffH > 2) {
        await createReminderIfMissing(m, "24H", "📣 Match Tomorrow", `${base}`, dk);
      } else if (diffH <= 2 && diffH > 0) {
        await createReminderIfMissing(m, "2H", "🚨 Match Soon (2 Hours)", `${base}`, dk);
      }

      // TODAY reminder (same date)
      if (dayKey(start) === dayKey(now)) {
        await createReminderIfMissing(m, "TODAY", "🏐 Match Today", `${base}`, dk);
      }
    }
  });
}

import express from "express";
import { requireRole } from "../utils/roles.js";

const router = express.Router();

// SAVE attendance lines + CLOSE
router.post("/sessions/:id/save", requireRole(["ADMIN","COACH","SUPER_ADMIN"]), async (req, res) => {
  const id = req.params.id;
  const { lines } = req.body; // [{playerId, present, note?}]

  const session = await req.db.attendanceSession.findUnique({ where: { id } });
  if (!session) return res.status(404).json({ ok:false, error:"Session not found" });

  if (session.status === "CLOSED") {
    return res.status(403).json({ ok:false, error:"Attendance closed" });
  }

  if (!Array.isArray(lines) || lines.length === 0) {
    return res.status(400).json({ ok:false, error:"Attendance lines required" });
  }

  for (const l of lines) {
    await req.db.attendanceLine.upsert({
      where: { attendanceSessionId_playerId: { attendanceSessionId: id, playerId: l.playerId } },
      update: { present: !!l.present, note: l.note || null },
      create: { attendanceSessionId: id, playerId: l.playerId, present: !!l.present, note: l.note || null },
    });
  }

  const updatedSession = await req.db.attendanceSession.update({
    where: { id },
    data: { status: "CLOSED", lockedAt: new Date().toISOString(), lockedBy: req.user.id },
  });

  res.json({ ok:true, session: updatedSession });
});

// EDIT session/lines: only SUPER_ADMIN if CLOSED
router.patch("/sessions/:id", requireRole(["ADMIN","COACH","SUPER_ADMIN"]), async (req, res) => {
  const id = req.params.id;
  const session = await req.db.attendanceSession.findUnique({ where: { id } });
  if (!session) return res.status(404).json({ ok:false, error:"Session not found" });

  if (session.status === "CLOSED" && req.user.role !== "SUPER_ADMIN") {
    return res.status(403).json({ ok:false, error:"Attendance closed" });
  }

  // For JSON store demo, only updates session meta here.
  // If you need line editing, reuse the same upsert logic from /save.
  const updated = await req.db.attendanceSession.update({
    where: { id },
    data: { ...req.body },
  });

  res.json({ ok:true, session: updated });
});

export default router;

import puppeteer from "puppeteer";

export async function htmlToPdfBuffer(html) {
  const browser = await puppeteer.launch({
    args: ["--no-sandbox","--disable-setuid-sandbox"],
  });
  try {
    const page = await browser.newPage();
    await page.setContent(html, { waitUntil: "networkidle0" });
    const pdf = await page.pdf({
      format: "A4",
      printBackground: true,
      margin: { top: "14mm", right: "12mm", bottom: "14mm", left: "12mm" },
    });
    return pdf;
  } finally {
    await browser.close();
  }
}

router.get("/status", async (req, res) => {

  const existing = await req.db.contractAcceptance.findFirst({ where: { userId, contractKey: CONTRACT_KEY } });
  res.json({
    ok: true,

    accepted: !!existing,
    acceptedAt: existing?.acceptedAt || null,
    acceptedBy: existing?.acceptedBy || null,
    contractKey: CONTRACT_KEY,
  });

router.post("/accept", async (req, res) => {
  const userId = req.user.id;
  const { accepterFullName, acceptedBy, guardianIdNumber, guardianPhoneNumber } = req.body || {};

  // find player profile (in this demo, userId == playerId is not implemented; adjust to your auth)
  // For now: if you store DOB on a user record, replace this lookup.
  // We'll try to find a player with id=userId; otherwise reject.
  const player = await req.db.player.findUnique({ where: { id: userId } });
  if (!player?.dob) return res.status(400).json({ ok:false, error:"Player DOB not found for age check." });

  const age = calcAge(player.dob);
  if (age === null) return res.status(400).json({ ok:false, error:"Invalid DOB" });

  const isMinor = age < 18;

  if (!accepterFullName || !String(accepterFullName).trim()) {
    return res.status(400).json({ ok:false, error:"Full name is required" });

    }

    if (acceptedBy !== "SELF") return res.status(400).json({ ok:false, error:"Adult must confirm as SELF" });

      userId,
      contractKey: CONTRACT_KEY,
      contractVersionHash: CONTRACT_VERSION_HASH,

import express from "express";
const router = express.Router();

function isMatchLocked(match) {
  return !!match.statsEntered || !!match.scoreLocked || match.scoreSource === "STATS";
}

// INIT: match + roster (photo/name/jersey)
router.get("/matches/:matchId/stats-touch/init", async (req, res) => {
  const { matchId } = req.params;
  const teamId = req.query.teamId;

  const match = await req.db.match.findUnique({ where: { id: matchId } });
  if (!match) return res.status(404).json({ ok:false, error:"Match not found" });

  const players = await req.db.player.findMany({
    where: { teamId },
    select: { id:true, firstName:true, lastName:true, jerseyNumber:true, photoUrl:true },
    orderBy: [{ jerseyNumber: "asc" }, { lastName: "asc" }],
  });

  res.json({ ok:true, match, players });
});

// POST EVENT
router.post("/matches/:matchId/events", async (req, res) => {
  const { matchId } = req.params;
  const { playerId, action, outcome, teamId } = req.body || {};

  const match = await req.db.match.findUnique({ where: { id: matchId } });
  if (!match) return res.status(404).json({ ok:false, error:"Match not found" });
  if (isMatchLocked(match)) return res.status(403).json({ ok:false, error:"Stats locked: already submitted." });

  if (!playerId || !action || !outcome) {
    return res.status(400).json({ ok:false, error:"Missing playerId/action/outcome" });
  }

  const event = await req.db.matchEvent.create({
    data: {
      matchId,
      teamId,
      playerId,
      action,
      outcome,
      createdAt: new Date().toISOString(),
      createdBy: req.user?.id || null,
    },
  });

  res.json({ ok:true, event, match });
});

// DELETE EVENT (UNDO)
router.delete("/matches/:matchId/events/:eventId", async (req, res) => {
  const { matchId, eventId } = req.params;

  const match = await req.db.match.findUnique({ where: { id: matchId } });
  if (!match) return res.status(404).json({ ok:false, error:"Match not found" });
  if (isMatchLocked(match)) return res.status(403).json({ ok:false, error:"Stats locked: already submitted." });

  await req.db.matchEvent.delete({ where: { id: eventId } });
  res.json({ ok:true, match });
});

export default router;

import React, { useEffect, useMemo, useState } from "react";

const ACTIONS = [
  { key: "SERVE", label: "Serve" },
  { key: "RECEIVE", label: "Receive" },
  { key: "SET", label: "Set" },
  { key: "ATTACK", label: "Attack" },
  { key: "BLOCK", label: "Block" },
  { key: "DIG", label: "Dig" },
  { key: "FREEBALL", label: "Free Ball" },
];

const OUTCOMES = [
  { key: "PLUS", label: "+ (Win)", hint: "Point / Success" },
  { key: "ZERO", label: "0 (OK)", hint: "Neutral" },
  { key: "MINUS", label: "- (Err)", hint: "Error / Lost point" },
];

export default function TouchStatsEntry({ matchId, teamId }) {
  const [loading, setLoading] = useState(true);
  const [match, setMatch] = useState(null);
  const [players, setPlayers] = useState([]);
  const [selectedPlayerId, setSelectedPlayerId] = useState(null);
  const [selectedAction, setSelectedAction] = useState(null);
  const [recent, setRecent] = useState([]);

  const selectedPlayer = useMemo(
    () => players.find((p) => p.id === selectedPlayerId),
    [players, selectedPlayerId]
  );

  const isLocked = !!match?.statsEntered || !!match?.scoreLocked || match?.scoreSource === "STATS";

  useEffect(() => {
    (async () => {
      try {
        setLoading(true);
        const r = await fetch(`/api/matches/${matchId}/stats-touch/init?teamId=${teamId}`);
        const data = await r.json();
        if (data?.ok) {
          setMatch(data.match);
          setPlayers(data.players || []);
          if (data.players?.[0]?.id) setSelectedPlayerId(data.players[0].id);
        }
      } finally {
        setLoading(false);
      }
    })();
  }, [matchId, teamId]);

  async function postEvent({ playerId, action, outcome }) {
    const tempId = `temp_${Date.now()}`;
    const uiItem = {
      id: tempId,
      at: new Date().toISOString(),
      playerId,
      playerName: `${selectedPlayer?.firstName || ""} ${selectedPlayer?.lastName || ""}`.trim(),
      jerseyNumber: selectedPlayer?.jerseyNumber ?? "",
      action,
      outcome,
    };

    setRecent((prev) => [uiItem, ...prev].slice(0, 12));

    const res = await fetch(`/api/matches/${matchId}/events`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ playerId, action, outcome, teamId }),
    });

    const data = await res.json().catch(() => null);
    if (!res.ok || !data?.ok) {
      setRecent((prev) => prev.filter((x) => x.id !== tempId));
      alert(data?.error || "Failed to save event.");
      return;
    }

    setRecent((prev) => prev.map((x) => (x.id === tempId ? { ...x, id: data.event.id } : x)));
    if (data.match) setMatch(data.match);
  }

  async function undoLast() {
    const last = recent[0];
    if (!last) return;

    const res = await fetch(`/api/matches/${matchId}/events/${last.id}`, { method: "DELETE" });
    const data = await res.json().catch(() => null);
    if (!res.ok || !data?.ok) {
      alert(data?.error || "Undo failed.");
      return;
    }

    setRecent((prev) => prev.slice(1));
    if (data.match) setMatch(data.match);
  }

  function outcomePill(outcome) {
    if (outcome === "PLUS") return "✅ +";
    if (outcome === "MINUS") return "❌ -";
    return "➖ 0";
  }

  if (loading) return <div style={S.wrap}>Loading stats...</div>;
  if (!match) return <div style={S.wrap}>Match not found.</div>;

  return (
    <div style={S.wrap}>
      <div style={S.headerRow}>
        <div>
          <div style={S.title}>Enter Stats (Touch)</div>
          <div style={S.sub}>
            Match: <b>{matchId}</b>{" "}
            {isLocked ? <span style={S.locked}>LOCKED</span> : <span style={S.open}>OPEN</span>}
          </div>
        </div>

        <button style={{ ...S.undoBtn, opacity: recent.length ? 1 : 0.4 }} disabled={!recent.length || isLocked} onClick={undoLast}>
          Undo Last
        </button>
      </div>

      <div style={S.playerGrid}>
        {players.map((p) => {
          const active = p.id === selectedPlayerId;
          return (
            <button key={p.id} style={{ ...S.playerCard, borderColor: active ? "#111" : "#ddd" }} onClick={() => setSelectedPlayerId(p.id)}>
              <img src={p.photoUrl || "/default-avatar.png"} alt="player" style={S.playerPhoto} />
              <div style={S.playerInfo}>
                <div style={S.playerName}>{(p.firstName || "").toUpperCase()} {(p.lastName || "").toUpperCase()}</div>
                <div style={S.playerMeta}>Jersey: <b>{p.jerseyNumber ?? "-"}</b></div>
              </div>
            </button>
          );
        })}
      </div>

      <div style={S.padWrap}>
        <div style={S.padTitle}>1) Choose Action</div>
        <div style={S.actionGrid}>
          {ACTIONS.map((a) => (
            <button
              key={a.key}
              style={{ ...S.actionBtn, background: selectedAction === a.key ? "#111" : "#fff", color: selectedAction === a.key ? "#fff" : "#111" }}
              disabled={isLocked}
              onClick={() => setSelectedAction(a.key)}
            >
              {a.label}
            </button>
          ))}
        </div>

        <div style={{ height: 10 }} />

        <div style={S.padTitle}>2) Choose Outcome</div>
        <div style={S.outcomeRow}>
          {OUTCOMES.map((o) => (
            <button
              key={o.key}
              style={S.outcomeBtn}
              disabled={isLocked || !selectedPlayerId || !selectedAction}
              onClick={() => postEvent({ playerId: selectedPlayerId, action: selectedAction, outcome: o.key })}
            >
              <div style={S.outcomeBig}>{o.label}</div>
              <div style={S.outcomeHint}>{o.hint}</div>
            </button>
          ))}
        </div>

        <div style={S.tip}>Tap <b>Player</b> → <b>Action</b> → <b>Outcome</b>. Fast during rallies.</div>
      </div>

      <div style={S.recentWrap}>
        <div style={S.padTitle}>Recent</div>
        {recent.length === 0 ? (
          <div style={{ opacity: 0.7 }}>No events yet.</div>
        ) : (
          <div style={S.recentList}>
            {recent.map((e) => (
              <div key={e.id} style={S.recentItem}>
                <div style={S.recentLeft}>
                  <div style={S.recentName}>{e.playerName} <span style={{ opacity: 0.7 }}>#{e.jerseyNumber}</span></div>
                  <div style={S.recentMeta}>{e.action} • {new Date(e.at).toLocaleTimeString()}</div>
                </div>
                <div style={S.recentRight}>{outcomePill(e.outcome)}</div>
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}
